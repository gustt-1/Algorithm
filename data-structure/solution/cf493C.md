# [cf493C] [Vasya and Basketball](https://codeforces.com/problemset/problem/493/C)

## 题目描述
有两个篮球队进行比赛。给你两个数组 `a, b` 代表这两个队伍的`投篮距离`，如果投篮距离`小于` $d$ ，那么得 `2` 分，`大于`则得 `3` 分。请你找到一个 `d` 使得总得分 `suma - sumb` 的`最大值`，只需要打印出 `suma : sumb` ，答案有多个只需要打印其中一个。

## 我的思路
**前后缀、排序**

> 我们考虑到如果假设的 `d` 对于两个队伍的得分顺序是没有任何影响的，从 $n$ `可能不等于` $m$ 也可以看出。那么我们可以对于 `a, b` 组合而成的数组进行`从小到大`进行排序，我们还需要记录那个值属于 $a/b$ ，所以用一个结构体记录。

> 不难发现，我们枚举当前 `a` 的值作为 `d` ，那么前面无论 `a, b` 都是`小于 d` 的，我们考虑**后面的** `a, b` 的数量，那么分差即为 $3 * (sufa - sufb) + 2 * (len(a) - sufa - len(b) + sufb)$ ，所以我们只需要维护一个`后缀值（前缀值）`即可。我们初始化 `d = 0` （ $a, b$ 都可以获得 `3` 分），后续方便更新答案。

package main

import (
	"bufio"
	. "fmt"
	"io"
	"os"
	"slices"
)

const inf = 0x3f3f3f3f

type (
	i8  = int8
	i64 = int64
	i32 = int32
	u64 = uint64
	u32 = uint32
	f32 = float32
	f64 = float64
)

func solve(in io.Reader, out io.Writer) {
	//var T int
	//for Fscan(in, &T); T > 0; T-- {
	//
	//}
	type pair struct {
		ty  byte
		val int
	}
	var n, m int
	Fscan(in, &n)
	a := make([]int, n)
	for i := range a {
		Fscan(in, &a[i])
	}
	Fscan(in, &m)
	t := make([]pair, n+m)
	b := make([]int, m)
	for i := range a {
		t[i] = pair{'a', a[i]}
	}
	for i := range b {
		Fscan(in, &b[i])
		t[n+i] = pair{'b', b[i]}
	}
	slices.SortFunc(t, func(a, b pair) int {
		if a.val == b.val {
			if a.ty == b.ty {
				return 0
			}
			if a.ty == 'a' {
				return -1
			}
			return 1
		}
		return a.val - b.val
	})
	ans1, ans2 := 3*n, 3*m
	cnta, cntb := 0, 0
	for _, x := range t {
		if x.ty == 'a' {
			cnta++
		} else {
			cntb++
		}

		sufa := 3*(n-cnta) + 2*cnta
		sufb := 3*(m-cntb) + 2*cntb

		if sufa-sufb > ans1-ans2 ||
			(sufa-sufb == ans1-ans2 && sufa > ans1) {
			ans1, ans2 = sufa, sufb
		}
	}
	Fprint(out, ans1, ":", ans2)
}

func abs(x int) int {
	if x < 0 {
		return -x
	}
	return x
}

func abs64(x i64) i64 {
	if x < 0 {
		return -x
	}
	return x
}

func gcd(x, y int) int {
	for y != 0 {
		x, y = y, x%y
	}
	return x
}

func max64(x, y i64) i64 {
	if x > y {
		return x
	}
	return y
}

func min64(x, y i64) i64 {
	if x < y {
		return x
	}
	return y
}

func main() {
	solve(bufio.NewReader(os.Stdin), os.Stdout)
}
